新华林问题验证

问题描述：

$$
对于质数p \ \ (p \neq 2,5,11)
$$

$$
必定存在\ \ p=x+y+z
$$

$$
使得\ \ x \ast y \ast z=w^3 (w \in N)
$$

代码：（范围：1~10000）

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <stack>
#include <set>
#include <cstring>
#include <conio.h>
#include <string>
#include <queue>
using namespace std;

typedef long long ll;

bool isprime[10005];
ll prime[5005];
ll top=1;

bool canfind(ll x)
{
	ll tx=ceil((long double)pow(x,(double)1/(double)3));
	// cout<<x<<" "<<tx<<endl;
	if(tx*tx*tx==x)
	{
		cout<<tx<<"^3		\t"<<x<<"		\t";
		return 1;
	}
	return 0;
}

int main(int argc, char const *argv[])
{
	freopen("write.txt","w",stdout);
	memset(isprime,1,sizeof(isprime));
	for(int i=2;i<10000;i++)
	{
		if(isprime[i])
			prime[top++]=i;
		for(int j=1;j<top;j++)
		{
			if(i*prime[j]>10000)
				break;
			isprime[i*prime[j]]=0;
			if(i%prime[j]==0)
				break;
		}
	}
		

	for(int i=1;i<top;i++)
	{
		ll k=prime[i]/3;
		bool ff=0;
		for(ll a=1;a<=k;a++)
		{
			if(ff)
				break;
			for(ll b=a;b<=(k<<1);b++)
			{
				ll cc=prime[i]-a-b;
				if(cc<0)break;
				if(canfind(a*b*cc))
				{
					cout<<prime[i]<<" = "<<a<<"+"<<b<<"+"<<cc<<endl;
					ff=1;
					break;
				}
			}
		}
		// getch();	
	}

	return 0;
}

更新：C++11 多线程

使用 #include <thread> 头文件，并加入编译选项 -std=c++11 ，注意mingw下编译无法通过，会提示你是否引用此头文件，但实际上无论是否引用总会编译错误，因此建议使用Linux原生gcc或者下载MinGW-W64编译版本。这次更新加入了20个线程对前5000个素数（实际素数大小大约到50000）进行分段处理，使得相同时间下运算时间大幅减少（就是cpu快要爆炸）。

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <stack>

#include <ctime>

#include <string>

void work(int l,int r)

	for(int i=l;i<=r;i++)
    {
        ll k=prime[i]/3;

        {

//    cout<<top<<endll;

	thread th4(work,751,1000);
	thread th5(work,1001,1250);

运行效果：（By AMD Ryzen7 2700X）

原程序下跑50000：248.60s
多线程下用时： 23.912s

缺陷

1.庞大的CPU占用率

2.乱序输出

正在尝试把所有结果统一存储，但这样还会加大内存负担，排序的用时也是不可接受的。此外，线程的插入顺序也有待改进，当前模式下必须等最慢的线程跑完才能结束进程，如果改进一下算法的话，应该有可能使每一个线程的结束时间相当，达到最大化效率。

更新：V3.0

发现线程是可以用循环来创建的，同时可以将前面的线程全部.detch分离出去，只让最后一个线程挂载在主程序内运行。效率可以进一步提升。

效率：9.620s

代码如下：V3.0

#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <stack>
#include <set>
#include <ctime>
#include <cstring>
#include <conio.h>
#include <string>
#include <queue>
#include <Windows.h>
#include <thread>
using namespace std;

typedef long long ll;

bool isprime[100005];
int prime[50005];
ll top=1,cnt=0;

bool canfind(ll x)
{
    ll tx=ceil((long double)pow(x,(double)1/(double)3));
    if(tx*tx*tx==x)
    {
        cout<<tx<<"^3        \t"<<x<<"        \t";
        return 1;
    }
    return 0;
}

void work(int l,int r)
{
	for(int i=l;i<=r;i++)
    {
        ll k=prime[i]/3;
        bool ff=0;
        for(ll a=1;a<=k;a++)
        {
            if(ff)
                break;
            for(ll b=a;b<=(k<<1);b++)
            {
                ll cc=prime[i]-a-b;
                if(cc<0)break;
                if(canfind(a*b*cc))
                {
                    cout<<prime[i]<<" = "<<a<<"+"<<b<<"+"<<cc<<endl;
                    cnt++;
                    ff=1;
                    break;
                }
            }
        }
    }
}

int main(int argc, char const *argv[])
{
    freopen("write.txt","w",stdout);
    memset(isprime,1,sizeof(isprime));
    for(int i=2;i<50000;i++)
    {
        if(isprime[i])
            prime[top++]=i;
        for(int j=1;j<top;j++)
        {
            if(i*prime[j]>50000)
                break;
            isprime[i*prime[j]]=0;
            if(i%prime[j]==0)
                break;
        }
    }
    
    // 100线程循环拆分
    for(int i=1;i<5000;i+=50)
    {
        thread a(work,i,i+50-1);
        if(i==4951)
            a.join();
        else
            a.detach();
    }
    // 总数（验证用）：
    cout<<cnt<<endl;

    cout<<"Working Time:  "<<(double)clock()/CLOCKS_PER_SEC<<endl;

    return 0;
}
